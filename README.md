import turtle

t = turtle.Turtle()

for c in ['teal', 'green', 'cyan', 'turquoise',
          'rainbow', 'black', 'purple', 'orange',
          'brown', 'red', 'yellow', 'gray', 'green',
          'white', 'aqua', 'black', 'purple', 'orange',
          'yellow', 'green', 'gray', 'purple', 'brown',
          'red',
          ]:
    t.color(c)
    t.forward(75)
    t.left(45)
